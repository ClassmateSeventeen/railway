class FastModel(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super(FastModel, self).__init__()
        self.resnet = models.resnet18(pretrained=True)
        self.fc = nn.Linear(input_dim, output_dim)

    def forward(self, x):
        x = self.resnet(x)
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x
